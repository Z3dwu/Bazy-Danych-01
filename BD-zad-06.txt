ZAD1
	SELECT AVG(pensja) AS 'srednia pensja' FROM EMPLOYEE;
	SELECT AVG(pensja) AS 'srednia pensja', COUNT(JOB_POSITION_ID) AS 'liczba praconikow' FROM EMPLOYEE GROUP BY JOB_POSITION_ID;
	SELECT AVG(2025-YEAR(BIRTH_DATE)) AS 'sredni wiek pracownikow', DEPARTMENT_ID FROM EMPLOYEE GROUP BY DEPARTMENT_ID;

ZAD2
	SELECT SUM(pensja) AS 'laczna pensja', DEPARTMENT_ID FROM EMPLOYEE GROUP BY DEPARTMENT_ID;
	SELECT AVG(pensja) AS 'srednia pensja', JOB_POSITION_ID, COUNT(JOB_POSITION_ID) AS 'ilosc' FROM EMPLOYEE GROUP BY JOB_POSITION_ID HAVING COUNT(JOB_POSITION_ID) >= 3 AND AVG(pensja) > 3000;
	SELECT COUNT(DISTINCT pensja) AS 'liczba roznych pensji', DEPARTMENT_ID FROM EMPLOYEE GROUP BY DEPARTMENT_ID HAVING COUNT(*) > 4 ORDER BY 'liczba roznych pensji';

ZAD3
	CREATE TABLE `DEPARTMENT` AS SELECT * FROM company_2025.`DEPARTMENT`;
	CREATE TABLE `JOB_POSITION` AS SELECT * FROM company_2025.`JOB_POSITION`;
	SELECT DEPARTMENT_NAME, COUNT(EMPLOYEE.EMPLOYEE_NAME) FROM DEPARTMENT INNER JOIN EMPLOYEE ON DEPARTMENT.DEPARTMENT_ID = EMPLOYEE.DEPARTMENT_ID GROUP BY DEPARTMENT_NAME;
	SELECT DEPARTMENT_NAME, CONCAT(EMPLOYEE.FIRST_NAME, ' ', EMPLOYEE.LAST_NAME) AS 'Imie i nazwisko' FROM DEPARTMENT INNER JOIN EMPLOYEE ON DEPARTMENT.DEPARTMENT_ID = EMPLOYEE.DEPARTMENT_ID ORDER BY DEPARTMENT_NAME;

ZAD4
	SELECT LAST_NAME, DEPARTMENT.DEPARTMENT_NAME, BIRTH_DATE FROM EMPLOYEE INNER JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID WHERE YEAR(BIRTH_DATE) BETWEEN 1990 AND 1999 ORDER BY DEPARTMENT.DEPARTMENT_NAME;
	SELECT CONCAT(FIRST_NAME, ' ', LAST_NAME) AS 'Imie i nazwisko' FROM EMPLOYEE INNER JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID WHERE DEPARTMENT.DEPARTMENT_NAME = 'Magazyn' ORDER BY BIRTH_DATE DESC LIMIT 5;
	SELECT CONCAT(C1.SHORT_NAME, ' - ', C2.SHORT_NAME) AS 'Nazwy klientow' FROM CLIENT C1 JOIN CLIENT C2 ON C1.CLIENT_ID + 5 = C2.CLIENT_ID ORDER BY C1.CLIENT_ID;

ZAD5
	SELECT JOB_POSITION_ID, AVG(pensja) FROM EMPLOYEE INNER JOIN DEPARTMENT ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID WHERE DEPARTMENT.DEPARTMENT_NAME NOT IN ('IT','HR') GROUP BY JOB_POSITION_ID HAVING COUNT(*) > 3;
	SELECT Temp.JOB_POSITION_ID, FIRST_NAME, LAST_NAME, BIRTH_DATE, DEPARTMENT_ID FROM EMPLOYEE INNER JOIN (SELECT JOB_POSITION_ID, MIN(BIRTH_DATE) AS 'najstarszy', MAX(BIRTH_DATE) AS 'najmlodszy' FROM EMPLOYEE GROUP BY JOB_POSITION_ID) Temp ON EMPLOYEE.JOB_POSITION_ID = Temp.JOB_POSITION_ID AND EMPLOYEE.BIRTH_DATE = Temp.najmlodszy OR EMPLOYEE.BIRTH_DATE = Temp.najstarszy ORDER BY Temp.JOB_POSITION_ID;
